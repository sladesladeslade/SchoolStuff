\section*{Question 1}
Figure 1 shows a 1D bar of uniform cross-section fixed at the
left-end. It has length $l=1$ m and section area $A=0.0002$ m$^2$.
It is subjected to a uniform body force $F_x$ and a concentrated 
force $F_s$ at the right end. The youngâ€™s modulus of the material
is $E=1.0*10^{10}$ N/m$^2$. Using analytical (exact) method,
obtain the distribution and the maximum value of the displacement,
strain and stress, for the following cases.

\begin{center}
    \includegraphics{fig}  
\end{center}

\subsection*{Part 1}
$F_x = 1000$ N/m, $F_s = 1000$ N

\subsubsection*{Boundary Conditions:}
@ $x=0$: $u=0$ m, $V=1000$ N \\\\
$\epsilon = \frac{\partial u}{\partial x} = \frac{F_s}{EA}$ \\
$\sigma = \frac{F_s}{A} = \epsilon E$ \\
$EA\frac{\partial^2 u}{\partial x^2} + F_x = 0$ \\
$u(x) = \int{\frac{-F_x}{EA}}\,dx = \frac{-F_x}{2EA}x^2 + C_1x
+ C_2$ \\\\
apply boundary condition: \\
$u(0) = 0 = C_2$ \\
$\epsilon = \frac{\partial u}{\partial x} = \frac{-F_x}{EA}x +
C_1$ \\
$\epsilon(l) = \frac{F}{EA} = \frac{-F_x}{EA}l + C_1$\\
$C_1 = \frac{F_s + F_xl}{EA}$ \\\\
plug constants back in: \\
$u(x) = \frac{-F_x}{2EA}x^2 + \frac{F_s + F_xl}{EA}x$\\\\
max displacement: \\
$u(l) = u(1) = \frac{-1000}{2(1*10^{10})(0.0002)}(1)^2 +
\frac{1000 + 1000(1)}{(1*10^{10})(0.0002)}(1) $\\
\fbox{$u_{max} = 0.00075$ m} \\
\begin{center}
    \includegraphics[scale=0.5]{disp}  
\end{center}
max stress: \\
$\sigma_{max} = \frac{F_{max}}{A} = \frac{1000 + 1000}{0.0002}$\\
\fbox{$\sigma_{max} = 10*10^6$ Pa} \\
\begin{center}
    \includegraphics[scale=0.5]{stress}  
\end{center}
max strain: \\
$\epsilon_{max} = \frac{\sigma_{max}}{E} = \frac{10*10^6}{1*10^{10}} $\\
\fbox{$\epsilon_{max} = 0.001$} \\
\begin{center}
    \includegraphics[scale=0.5]{strain}  
\end{center}

\subsection*{Part 2}
$F_x = (100x + 1000)$ N/m, $F_s = 0$ N \\\\
using same equation for displacement: \\
$u(x) = \frac{-F_x}{2EA}x^2 + \frac{0 + F_xl}{EA}x$ \\
$u(l) = u(1) = \frac{-(100(1) + 1000)}{2(1*10^{10})(0.0002)}(1)^2 +
\frac{(100(1) + 1000)(1)}{(1*10^{10})(0.0002)}(1) $ \\
